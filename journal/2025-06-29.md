	- #+BEGIN_QUERY
	  {
	   :inputs [:-13d :+7d]
	   :query [:find (pull ?b [*])
	   :in $ ?start ?end
	   :where
	     (or-join [?b ?d]
	       (or
	         [?b :block/scheduled ?d]
	         [?b :block/deadline ?d]
	       )
	       (and
	         [?b :block/marker ?marker]
	         [(contains? #{"NOW" "LATER" "TODO"} ?marker)]
	         [?b :block/page ?p]
	         [?p :block/journal-day ?d]
	       )
	     )
	     [(> ?d ?start)]
	     [(< ?d ?end)]
	   ]
	  }
	  #+END_QUERY
